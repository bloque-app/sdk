      #Organizations

      Learn how to create and manage organizations using the Bloque SDK.

      #Overview

      Organizations are the foundation of the Bloque platform. They can be either businesses or individuals and contain all the necessary profile and compliance information.

      #Creating an Organization

      #Business Organization

      import { SDK } from '@bloque/sdk';
import type { CreateOrgParams } from '@bloque/sdk/orgs';

const bloque = new SDK({
  origin: 'your-origin',
  auth: {
    type: 'apiKey',
    apiKey: process.env.BLOQUE_API_KEY!,
  },
  mode: 'production',
});

const params: CreateOrgParams = {
  org_type: 'business',
  profile: {
    legal_name: 'Acme Corporation',
    tax_id: '123456789',
    incorporation_date: '2020-01-01',
    business_type: 'llc',
    incorporation_country_code: 'US',
    incorporation_state: 'CA',
    address_line1: '123 Main St',
    postal_code: '94103',
    city: 'San Francisco',
  },
  metadata: {
    source: 'web_app',
    campaign: 'q1_2024',
  },
};

const organization = await bloque.orgs.create(params);

      #Response

      #Organization

      interface Organization {
  urn: string;                           // Unique resource name
  org_type: 'business' | 'individual';   // Organization type
  profile: OrgProfile;                   // Organization profile
  metadata?: Record<string, unknown>;    // Custom metadata
  status: OrgStatus;                     // Organization status
}

      #Members

      Manage organization members.

      // List members
const members = await bloque.orgs.listMembers('bloque:org:abc123');

// Update a member's role
await bloque.orgs.members.update('bloque:member:m1', {
  role: 'admin',
});

// Remove a member
await bloque.orgs.members.remove('bloque:org:abc123', 'bloque:member:m1');
      MethodParametersDescription
      listMembers(orgUrn)stringList all members of the org
      members.update(urn, params)URN + { role }Update member role
      members.remove(orgUrn, memberUrn)Two URNsRemove member from org

      #Teams

      Organize members into teams with specific roles.

      // List teams
const teams = await bloque.orgs.teams.list('bloque:org:abc123');

// Update a team
await bloque.orgs.teams.update('bloque:team:t1', { name: 'Engineering' });

// List team members
const teamMembers = await bloque.orgs.teams.listMembers('bloque:team:t1');

// Update a team member's role
await bloque.orgs.teams.updateMember('bloque:team:t1', 'bloque:member:m1', {
  role: 'lead',
});

// Remove from team
await bloque.orgs.teams.removeMember('bloque:team:t1', 'bloque:member:m1');
      MethodDescription
      teams.list(orgUrn)List all teams in the org
      teams.update(teamUrn, params)Update team name or metadata
      teams.listMembers(teamUrn)List members of a team
      teams.updateMember(teamUrn, memberUrn, params)Update a member's team role
      teams.removeMember(teamUrn, memberUrn)Remove a member from the team

      #Invites

      Send, accept, reject, and manage organization invitations.

      // Create an invite
const invite = await bloque.orgs.invites.create('bloque:org:abc123', {
  email: 'new-member@example.com',
  role: 'member',
});

// List invites
const invites = await bloque.orgs.invites.list({ orgUrn: 'bloque:org:abc123' });

// Accept an invite
await bloque.orgs.invites.accept('invite-code-123');

// Reject an invite
await bloque.orgs.invites.reject('invite-code-123', 'Not interested');

// Resend an invite
await bloque.orgs.invites.resend('invite-code-123');
      MethodDescription
      invites.create(orgUrn, params)Send an invite
      invites.get(code)Get invite details by code
      invites.list(params?)List invites (filter by orgUrn, status)
      invites.accept(code)Accept an invite
      invites.reject(code, reason?)Reject an invite
      invites.resend(code)Resend an invite email

      #Best Practices

      1. Validate Data: Ensure all required fields are present before calling the API
      2. Handle Errors: Always use try-catch blocks
      3. Store URNs: Save the organization URN for future operations
      4. Use Metadata: Track additional context using metadata
      5. Test in Sandbox: Always test with mode: 'sandbox' first
